\documentclass[12pt]{article}
\usepackage{usecases}
\begin{document}
\begin{usecase}

\addtitle{Use Case 3}{Create Activities} 

%Scope: the system under design
\addfield{Scope:}{System-wide}	
%Primary Actor: Calls on the system to deliver its services.
\addfield{Primary Actor:}{Project Leader}

%Stakeholders and Interests: Who cares about this use case and what do they want?
\additemizedfield{Stakeholders and Interests:}{
	\item Project Leader: Needs to be able to create new activities for developers
}
\addfield{Description:}{Makes it so Developers and Project leader can create new activities}
%Main Success Scenario: A typical, unconditional happy path scenario of success.
\addscenario{Main Scenario:}{
	\item Project Leader creates new activity with name, discription and allocated work hours 
}

%Extensions: Alternate scenarios of success or failure.
\addscenario{Alternative Scenarios:}{
	\item[1.a] Project leader instantiates activity with null name:
		\begin{enumerate}
		\item[1.] System sets name to default name.
		\end{enumerate}
	\item[1.b] Project leader instantiates activity with null description:
		\begin{enumerate}
		\item[1.] Activity created with null discription.
		\end{enumerate}
	\item[1.c] Project leader instantiates activity with non positive allocated work hours:
		\begin{enumerate}
		\item[1.] System sets allocatedWorkHours to default allocatedWorkHours.
		\end{enumerate}
}
\end{usecase}
