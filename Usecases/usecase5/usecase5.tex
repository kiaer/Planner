\begin{usecase}

\addtitle{Use Case 5}{Define Developers} 

%Scope: the system under design
\addfield{Scope:}{System-wide}

%Primary Actor: Calls on the system to deliver its services.
\addfield{Primary Actor:}{Admin}

%Stakeholders and Interests: Who cares about this use case and what do they want?
\additemizedfield{\textsl{Administrator}:}{
	\item Administrator: Create and modifies developers.
}

%Preconditions: What must be true on start and worth telling the reader?
\addfield{Description:}{Makes it so admin can manage all information in developers, add and remove new developers.}
%when multiple
%\additemizedfield{Preconditions:}{} 

%Main Success Scenario: A typical, unconditional happy path scenario of success.
\addscenario{Main Success Scenario:}{
	\item Admin creates developer.
	\item Admin modifies developer.
}

%Extensions: Alternate scenarios of success or failure.
\addscenario{Alternative Scenarioes:}{
	\item[1.a] Admin instantiates user with null username:
		\begin{enumerate}
		\item[1.] Username is set to default username.
		\end{enumerate}
	\item[1.b] Admin instantiates user with null password:
		\begin{enumerate}
		\item[1.] Password is set to null value.
		\end{enumerate}
	\item[1.c] Admin instantiates user with null email:
		\begin{enumerate}
		\item[1.] Email is set to null value.
		\end{enumerate}
	\item[2.a] Admin sets username to null:
		\begin{enumerate}
		\item[1.] System shows failure message.
		\item[2.] System returns to step 2.
		\end{enumerate}
	\item[2.b] Admin sets password to null:
		\begin{enumerate}
		\item[1.] Password is set to null value.
		\end{enumerate}
	\item[2.c] Admin sets email to null:
		\begin{enumerate}
		\item[1.] Email is set to null value.
		\end{enumerate}
}
\end{usecase}
\end{document}
